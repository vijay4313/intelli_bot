344  inline
345  void setScaledRotationMatrix
346  (const Transformation & sR) {
347  Transformation squared_sR = sR*sR.transpose();
348  Scalar squared_scale
349  = static_cast<Scalar>(1./3.)
350  *(squared_sR(0,0)+squared_sR(1,1)+squared_sR(2,2));
351  if (squared_scale <= static_cast<Scalar>(0)) {
352  throw ScaleNotPositive();
353  }
354  Scalar scale = std::sqrt(squared_scale);
355  if (scale <= static_cast<Scalar>(0)) {
356  throw ScaleNotPositive();
357  }
358  quaternion() = sR/scale;
359  quaternion().coeffs() *= scale;
360  }
