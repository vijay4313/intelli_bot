167  inline
168  void fastMultiply(const SO2Group<Scalar>& other) {
169  Scalar lhs_real = unit_complex().x();
170  Scalar lhs_imag = unit_complex().y();
171  const Scalar & rhs_real = other.unit_complex().x();
172  const Scalar & rhs_imag = other.unit_complex().y();
173  // complex multiplication
174  unit_complex_nonconst().x() = lhs_real*rhs_real - lhs_imag*rhs_imag;
175  unit_complex_nonconst().y() = lhs_real*rhs_imag + lhs_imag*rhs_real;
176  }

204  inline
205  void normalize() {
206  Scalar length =
207  std::sqrt(unit_complex().x()*unit_complex().x()
208  + unit_complex().y()*unit_complex().y());
209  if(length < SophusConstants<Scalar>::epsilon()) {
210  throw SophusException("Complex number is (near) zero!");
211  }
212  unit_complex_nonconst().x() /= length;
213  unit_complex_nonconst().y() /= length;
214  }
